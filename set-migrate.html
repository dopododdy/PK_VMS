📁 ตาราง clinic_table_db (ข้อมูลผู้ป่วย)

✅ คัดลอกแล้ว!
CREATE TABLE IF NOT EXISTS clinic_table_db ( id BIGINT PRIMARY KEY GENERATED ALWAYS AS IDENTITY, rec_id VARCHAR(255) UNIQUE, animal_id VARCHAR(255), animal_name VARCHAR(255), animal_spp VARCHAR(255), animal_sex VARCHAR(50), animal_tag VARCHAR(255), owner_name VARCHAR(255), animal_dob DATE, phone_no VARCHAR(50), medrec TEXT, created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP, updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ); -- แก้ไข phone_no สำหรับฐานข้อมูลเดิมที่สร้างด้วย VARCHAR(20) ALTER TABLE IF EXISTS clinic_table_db ALTER COLUMN phone_no TYPE VARCHAR(50); -- เพิ่มคอลัมน์ animal_sex สำหรับฐานข้อมูลที่สร้างก่อนเวอร์ชันนี้ ALTER TABLE IF EXISTS clinic_table_db ADD COLUMN IF NOT EXISTS animal_sex VARCHAR(50);

💊 ตาราง pharmacy (ยากิน / ยาฉีด)

✅ คัดลอกแล้ว!
CREATE TABLE IF NOT EXISTS pharmacy ( id UUID PRIMARY KEY DEFAULT gen_random_uuid(), item_name TEXT NOT NULL, name_th TEXT, category TEXT, sub_category TEXT, unit TEXT DEFAULT 'tablet', concentration_mg_unit NUMERIC(10,4) DEFAULT 1, price_per_unit NUMERIC(10,2) DEFAULT 0, bottle_volume_ml NUMERIC(10,2) DEFAULT 0, side_effects TEXT, is_medication BOOLEAN DEFAULT TRUE, dog_dose_1 NUMERIC(10,4), dog_for_1 TEXT, dog_dose_2 NUMERIC(10,4), dog_for_2 TEXT, dog_dose_3 NUMERIC(10,4), dog_for_3 TEXT, cat_dose_1 NUMERIC(10,4), cat_for_1 TEXT, cat_dose_2 NUMERIC(10,4), cat_for_2 TEXT, cat_dose_3 NUMERIC(10,4), cat_for_3 TEXT, created_at TIMESTAMPTZ DEFAULT NOW(), updated_at TIMESTAMPTZ DEFAULT NOW() );

เตรียมไฟล์ข้อมูลผู้ป่วย

รองรับไฟล์ Excel (.xlsx) และ XML จาก Microsoft Access

📊 ตัวเลือก ก — ไฟล์ Excel (.xlsx)

รองรับทั้งรูปแบบภาษาอังกฤษ (mainDB.xlsx) และรูปแบบภาษาไทย (Old_DB.xlsx):

รูปแบบภาษาอังกฤษ (mainDB.xlsx) HN → rec_id + owner HNAnimalName → animal_name AnimalSpp → animal_sppAnimalSex → animal_sex SpecialNote → animal_tagOwnerName → owner_name AnimalDOB → animal_dobPhoneNo → phone_no MedicalHistory → medrec รูปแบบภาษาไทย (Old_DB.xlsx) HN → rec_idชื่อสัตว์ → animal_name ชนิด - เพศ → animal_sppข้อควรระวัง → animal_tag ชื่อ เจ้าของ → owner_nameวดป เกิด → animal_dob เบอร์โทรศัพท์ → phone_noPE ตรวจร่างกาย → medrec

💡 ถ้า HN ซ้ำกัน ระบบจะทับข้อมูลเดิมโดยอัตโนมัติ

🗄️ ตัวเลือก ข — XML จาก Microsoft Access

  1. เปิดไฟล์ ClinicDB.mdb ใน Microsoft Access
  2. คลิกขวาที่ตาราง ClinicTableDB → Export → XML File
  3. เลือกส่งออก Data (XML) เท่านั้น
  4. บันทึกเป็นไฟล์ .xml
